on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { NOT = { has_country_flag = has_starting_oob } }
				d_create_army = yes
			}
			if = {
				limit = { has_dlc = "Man the Guns" }
				every_country = {
					set_technology = {
						early_ship_hull_light = 1
						popup = no
					}
					set_technology = {
						basic_torpedo = 1
						popup = no
					}
					set_technology = {
						early_ship_hull_submarine = 1
						popup = no
					}
				}
				else = {
					every_country = {
						set_technology = {
							early_destroyer = 1
							popup = no
						}
						set_technology = {
							early_submarine = 1
							popup = no
						}
					}
				}
			}

			RUS = { add_to_array = { global.Cold_War_Factions = THIS } }
			CHI = { add_to_array = { global.Cold_War_Factions = THIS } }

			ZZZ = { do_cold_war_sphere_arrays = yes }
		}
	}

	
	on_puppet = { # Yes
		effect = {
			log = "Puppeted: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			puppet_command = yes
		}
	}
	on_release_as_puppet = { # Yes
		effect = {
			log = "Released as Puppet: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			puppet_command = yes
		}
	}
	on_annex = { # Yes
		effect = {
			log = "Annexed: [FROM.GetNameWithFlag] ([FROM.GetTag])"
			FROM = {
				drop_cosmetic_tag = yes
			}
		}
	}
	on_liberate = {
		effect = {
			log = "Liberated: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			Random_Cosmetic = yes
		}
	}
	on_release_as_free = { # No
		effect = {
			log = "Released as free: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			Random_Cosmetic = yes
		}
	}
	on_capitulation = { # Yes
		effect = {
			log = "Capitulation: [FROM.GetNameWithFlag] defeats [ROOT.GetNameWithFlag]"
		}
	}
	on_send_volunteers = {
		effect = {
			log = "Volunteers: [ROOT.GetNameWithFlag] to [FROM.GetNameWithFlag]"
			if = {
				limit = {
					ROOT = { tag = CHI }
					FROM = { has_civil_war = yes new_marxist_or_hypersoc_without_alb = yes }
				}
				if = {
					limit = {
						NOT = {
							is_in_array = { ROOT.friendship_treaties = FROM }
						}
					}
					ROOT = { save_event_target_as = treaty_target }
					FROM = { country_event = Treaties.1 }
				}
			}
		}
	}
	on_declare_war = {
		effect = {
			log = "Declaration: [ROOT.GetNameWithFlag] ([ROOT.GetTag]) declares on [FROM.GetNameWithFlag] ([FROM.GetTag])"
			
			ROOT = {
				if = {
					limit = {
						civilwar_target = FROM
					}
					log = "^^ Civil War"
				}
				add_to_array = { global.civil_wars = FROM }
			}

			if = {
				limit = {
					ROOT = {
						tag = GER
					}
				}
				# KRAUTS KRAUTS KRAUTS KRAUTS KRAUTS KRAUTS KRAUTS
				FROM = { save_event_target_as = german_target }
				FRA = {
					country_event = { id = FRA_Germany.1 hours = 12 }
				}
			}
		}	
	}
	on_civil_war_end = {
		effect = {
			log = "Civil War End: [ROOT.GetNameWithFlag] ([ROOT.GetTag]) defeats on [FROM.GetNameWithFlag] ([FROM.GetTag])"
			ROOT = {
				remove_from_array = { global.civil_wars = ROOT } 
				remove_from_array = { global.civil_wars = FROM } 
			}
		}
	}
	on_new_term_election = {
		random_events = {
			100 = ALB_Congress.1
		}
	}

	on_weekly = {
		effect = {
			ZZZ = { do_cold_war_sphere_arrays = yes }
		}
	}
}
