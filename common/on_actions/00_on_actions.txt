on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { NOT = { has_country_flag = has_starting_oob } }
				d_create_army = yes
			}
			if = {
				limit = { has_dlc = "Man the Guns" }
				every_country = {
					set_technology = {
						early_ship_hull_light = 1
						popup = no
					}
					set_technology = {
						basic_torpedo = 1
						popup = no
					}
					set_technology = {
						early_ship_hull_submarine = 1
						popup = no
					}
				}
				else = {
					every_country = {
						set_technology = {
							early_destroyer = 1
							popup = no
						}
						set_technology = {
							early_submarine = 1
							popup = no
						}
					}
				}
			}

			RUS = { add_to_array = { global.Cold_War_Factions = THIS } }
			PRC = { add_to_array = { global.Cold_War_Factions = THIS } }
		}
	}

	
	on_puppet = { # Yes
		effect = {
			log = "Puppeted: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			puppet_command = yes
		}
	}
	on_release_as_puppet = { # Yes
		effect = {
			log = "Released as Puppet: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			puppet_command = yes
		}
	}
	on_annex = { # Yes
		effect = {
			log = "Annexed: [FROM.GetNameWithFlag] ([FROM.GetTag])"
			FROM = {
				drop_cosmetic_tag = yes
			}
		}
	}
	on_liberate = {
		effect = {
			log = "Liberated: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			Random_Cosmetic = yes
		}
	}
	on_release_as_free = { # No
		effect = {
			log = "Released as free: [ROOT.GetNameWithFlag] ([ROOT.GetTag])"
			Random_Cosmetic = yes
		}
	}
	on_capitulation = { # Yes
		effect = {
			log = "Capitulation: [FROM.GetNameWithFlag] defeats [ROOT.GetNameWithFlag]"
		}
	}
	on_declare_war = {
		effect = {
			log = "Declaration: [ROOT.GetNameWithFlag] ([ROOT.GetTag]) declares on [FROM.GetNameWithFlag] ([FROM.GetTag])"
			
			ROOT = {
				if = {
					limit = {
						civilwar_target = FROM
					}
					log = "^^ Civil War"
				}
			}
		}	
	}
	on_new_term_election = {
		random_events = {
			100 = ALB_Congress.1
		}
	}
}
