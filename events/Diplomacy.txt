country_event = {
	id = Puppet_Event.1
	title = Puppet_Event.1.TITLE
	desc = Puppet_Event.1.DESC
	
	is_triggered_only = yes
	immediate = { log = "[GetDateText]: [Root.GetName]: Event Puppet_Event.1 ([new_puppet.GetName])" }
	
	option = {
		name = Puppet_Event.1.A # AUTONOMOUS
		log = "[GetDateText]: [Root.GetName]: Event Puppet_Event.1 - Puppet_Event.1.A"
		ai_chance = {
			base = 20

			modifier = {
				factor = 2.5
				original_tag = RUS
				event_target:new_puppet = {
					capital_scope = {
						is_on_continent = europe
					}
				}
			}

			modifier = {
				factor = 1.5
				any_country = {
					if = {
						limit = {
							OR = {
								tag = PREV
								AND = {
									is_subject_of = PREV
									has_autonomy_state = autonomy_autonomous_zone
								}
							}
						}
						is_owner_neighbor_of = event_target:new_puppet
					}
				}
			}
			
			modifier = {
				factor = 100
				original_tag = RUS
				event_target:new_puppet = {
					SOV_me_likey_ssr = yes
				}
			}
		}
		set_autonomy = {
			target = event_target:new_puppet
			autonomy_state = autonomy_autonomous_zone
		}
		event_target:new_puppet = {
			puppet_lose_military = yes
			puppet_autonomous = yes
		}
	}
	
	option = {
		name = Puppet_Event.1.B.A
		log = "[GetDateText]: [Root.GetName]: Event Puppet_Event.1 - Puppet_Event.1.B.A"
		ai_chance = {
			base = 5
		}
		set_autonomy = {
			target = event_target:new_puppet
			autonomy_state = autonomy_autonomous_zone
		}
		event_target:new_puppet = {
			puppet_military = yes
		}
	}
	
	option = {
		name = Puppet_Event.1.C # SATELLITE
		log = "[GetDateText]: [Root.GetName]: Event Puppet_Event.1 - Puppet_Event.1.C"
		ai_chance = {
			base = 50
		}
		set_autonomy = {
			target = event_target:new_puppet
			autonomy_state = autonomy_satellite
		}
		event_target:new_puppet = {
			puppet_lose_military = yes
			puppet_satellite = yes
		}
	}
}
